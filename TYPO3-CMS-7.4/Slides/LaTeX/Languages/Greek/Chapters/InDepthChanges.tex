	%\lstset{basicstyle=\tiny\ttfamily}

	\begin{itemize}

		\item \greektext Η μορφοποίηση μεγέθους αρχείου υποστηρίζει δύο λέξεις\latintext-\greektextκλειδιά
		επιπλέον στη λίστα των ετικετών τώρα:

			\latintext \begin{itemize}
				\item \small\texttt{iec} \greektext(προεπιλογή)\newline
					\small \greektext (δύναμη του 2, ετικέτες: \latintext \texttt{| Ki| Mi| Gi| Ti| Pi| Ei| Zi| Yi})\normalsize
				\item \small\texttt{si}\newline
					\small \greektext (δύναμη του 10, ετικέτες: \latintext \texttt{| k| M| G| T| P| E| Z| Y})\normalsize
			\end{itemize}

		\item \greektext Η μορφοποίηση τίθεται στην \latintext TypoScript \greektext για παράδειγμα:\newline
			\latintext \texttt{bytes.labels = iec}

		\latintext \begin{lstlisting}
			echo GeneralUtility::formatSize(85123);
			// => prin "83.1 K"
			// => meta "83.13 Ki"
		\end{lstlisting}

	\end{itemize}
