\begin{frame}[fragile]
	\frametitle{Fonctions dépréciées et retirées}
	\framesubtitle{Méthodes et constantes liées à CLI}

	% decrease font size for code listing
	%\lstset{basicstyle=\tiny\ttfamily}

	\begin{itemize}

		\item Les opérations concernant l'exécution CLI usuelle avec l'option CLIkey ont
			été déplacées dans le \texttt{CliRequestHandler} et donc, les méthodes suivantes sont
			marquées \textbf{dépréciées}~:

			\begin{lstlisting}
				BackendUserAuthentication->checkCLIuser()
			\end{lstlisting}

		\item Les constantes et paramètres globaux suivants ont aussi été marqués
			 \textbf{dépréciés}~:

			\begin{lstlisting}
				const TYPO3_cliKey
				const TYPO3_cliInclude
				$GLOBALS['MCONF']['name']
				$GLOBALS['temp_cliScriptPath']
				$GLOBALS['temp_cliKey']
			\end{lstlisting}

	\end{itemize}

\end{frame}

\begin{frame}[fragile]
	\frametitle{Fonctions dépréciées et retirées}
	\framesubtitle{IconUtility}

	%\lstset{basicstyle=\tiny\ttfamily}

	\begin{itemize}

		\item L'interface \texttt{IconUtilityOverrideResourceIconHookInterface}\newline

		\item Toutes les méthodes publiques de la classe sont marquées \textbf{dépréciées}~:

					\tabto{1cm}\texttt{IconUtility::getSpriteIconForRecord()}\newline
					\tabto{1cm}\texttt{IconUtility::getSpriteIconClasses()}

			\begin{column}{.5\textwidth}
				\vspace{-0.18cm}\tiny
					\texttt{IconUtility::getIcon()}\newline
					\texttt{IconUtility::getSpriteIconForFile()}\newline
					\texttt{IconUtility::getSpriteIconForResource()}\newline
			\end{column}
		\end{columns}

		\vspace{0.2cm}
		\normalsize

			\small
				(certaines méthodes sont déplacées dans \texttt{IconFactory})
			\normalsize

		\item L'option TSconfig de page suivante a aussi été marquée \textbf{dépréciée}~:

			\begin{lstlisting}
				mod.wizards.newContentElement.wizardItems.*.elements.*.icon
			\end{lstlisting}

	\end{itemize}

\begin{frame}[fragile]
	\frametitle{Fonctions dépréciées et retirées}
	\framesubtitle{Extension système \texttt{form}}

	\begin{itemize}

		\item L'extension système \texttt{form} se base maintenant sur Extbase/Fluid

		\item Ainsi, toutes les configurations TypoScript de l'apparence sont marquées
			\textbf{dépréciées} et l'utilisation de \texttt{layout} \underline{n'}est
			\underline{plus} recommandée~:

			\begin{lstlisting}
				10 = FORM
				10 {
				  layout {
				    containerWrap = <div><elements ></elements></div>
				    elementWrap = <div><element ></element></div>
				  }
				}
			\end{lstlisting}

	\end{itemize}

\end{frame}
