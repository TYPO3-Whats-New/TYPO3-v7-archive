\begin{frame}[fragile]
	\frametitle{Funzionalità deprecate/rimosse}
	\framesubtitle{Costanti e metodi CLI collegati}

	% decrease font size for code listing
	%\lstset{basicstyle=\tiny\ttfamily}

	\begin{itemize}

		\item La logica per quanto riguarda regolari script basati su CLI con l'opzione CLIkey è stato spostato
			in \texttt{CliRequestHandler} e quindi, il seguente metodo è stato contrassegnato come
			 \textbf{deprecated}:

			\begin{lstlisting}
				BackendUserAuthentication->checkCLIuser()
			\end{lstlisting}

		\item Inoltre, le seguenti costanti e parametri globali sono stati segnati come
			 \textbf{deprecati}:

			\begin{lstlisting}
				const TYPO3_cliKey
				const TYPO3_cliInclude
				$GLOBALS['MCONF']['name']
				$GLOBALS['temp_cliScriptPath']
				$GLOBALS['temp_cliKey']
			\end{lstlisting}

	\end{itemize}

\end{frame}

\begin{frame}[fragile]
	\frametitle{Funzionalità deprecate/rimosse}
	\framesubtitle{IconUtility}

	% decrease font size for code listing
	%\lstset{basicstyle=\tiny\ttfamily}

	\begin{itemize}

		\item L'interfaccia \texttt{IconUtilityOverrideResourceIconHookInterface}\newline
			come anche la classe \texttt{IconUtility} saranno rimosse in TYPO3 CMS 8

		\item Tutti i metodi pubblici di questa classe sono stati segnati come \textbf{deprecati}:

		\vspace{0.2cm}

		\begin{columns}[T]
			\begin{column}{.5\textwidth}
				\tiny
					\tabto{1cm}\texttt{IconUtility::skinImg()}\newline
					\tabto{1cm}\texttt{IconUtility::getSpriteIcon()}\newline
					\tabto{1cm}\texttt{IconUtility::getSpriteIconForRecord()}\newline
					\tabto{1cm}\texttt{IconUtility::getSpriteIconClasses()}
			\end{column}
			\begin{column}{.5\textwidth}
				\vspace{-0.18cm}\tiny
					\texttt{IconUtility::getIcon()}\newline
					\texttt{IconUtility::getSpriteIconForFile()}\newline
					\texttt{IconUtility::getSpriteIconForResource()}\newline
			\end{column}
		\end{columns}

		\vspace{0.2cm}
		\normalsize

			\small
				(alcuni metodi sono stati spostati in \texttt{IconFactory})
			\normalsize

		\item Inoltre, le seguenti impostazioni PageTSConfig sono state marcate come \textbf{deprecate}:

			\begin{lstlisting}
				mod.wizards.newContentElement.wizardItems.*.elements.*.icon
			\end{lstlisting}

	\end{itemize}

\end{frame}

\begin{frame}[fragile]
	\frametitle{Funzionalità deprecate/rimosse}
	\framesubtitle{Estensione di sistema \texttt{form}}

	\begin{itemize}

		\item L'estensione di sistema \texttt{form} è ora basata su Extbase/Fluid

		\item Pertanto tutte le impostazioni di layout basate su TypoScript sono state segnate come
			\textbf{deprecate} e l'utilizzo di \texttt{layout} \underline{non}
			è più raccomandato:

			\begin{lstlisting}
				10 = FORM
				10 {
				  layout {
				    containerWrap = <div><elements ></elements></div>
				    elementWrap = <div><element ></element></div>
				  }
				}
			\end{lstlisting}

	\end{itemize}

\end{frame}
